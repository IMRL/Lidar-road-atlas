# IDE folder in VS, Xcode etc. CMake 3.12+, older versions have only the FOLDER
# property that would have to be set on each target separately.
set(CMAKE_FOLDER "Corrade/Containers")

set(CorradeContainers_HEADERS
    AnyReference.h
    Array.h
    ArrayTuple.h
    ArrayView.h
    ArrayViewStl.h
    ArrayViewStlSpan.h
    BigEnumSet.h
    BigEnumSet.hpp
    constructHelpers.h
    Containers.h
    EnumSet.h
    EnumSet.hpp
    GrowableArray.h
    initializeHelpers.h
    LinkedList.h
    MoveReference.h
    Optional.h
    OptionalStl.h
    Pair.h
    PairStl.h
    Pointer.h
    PointerStl.h
    Reference.h
    ScopeGuard.h
    sequenceHelpers.h
    StaticArray.h
    StridedArrayView.h
    StridedArrayViewStl.h
    String.h
    StringStl.h
    StringStlView.h
    StringView.h
    Triple.h
    TripleStl.h)

set(CorradeContainers_PRIVATE_HEADERS
    Implementation/RawForwardList.h)

if(CORRADE_BUILD_DEPRECATED)
    list(APPEND CorradeContainers_HEADERS Tags.h)
endif()

# Sources are compiled as part of Utility

# Force IDEs to display all header files in project view
add_custom_target(CorradeContainers-headers SOURCES
    ${CorradeContainers_HEADERS}
    ${CorradeContainers_PRIVATE_HEADERS})

install(FILES ${CorradeContainers_HEADERS} DESTINATION ${CORRADE_INCLUDE_INSTALL_DIR}/Containers)

if(BUILD_TESTS)
    add_subdirectory(Test)
endif()

# Corrade::Containers target alias for superprojects
add_library(CorradeContainers INTERFACE)
target_link_libraries(CorradeContainers INTERFACE CorradeUtility)
add_library(Corrade::Containers ALIAS CorradeContainers)
